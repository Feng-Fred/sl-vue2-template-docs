import { defineComponent } from "vue";
import { Icon, Typography } from "sl-vue2-template";
import { type IconName } from "sl-vue2-template/lib/cjs/types/Icon/Icon";
import "./styles.scss";

const { Text } = Typography;

export default defineComponent({
  data() {
    return {
      show: false,
    };
  },
  methods: {
    onClose() {
      this.show = !this.show;
    },
  },
  render() {
    return (
      <div>
        <ul>
          {List.map((item) => {
            return (
              <li>
                <Icon name={item} />
                <Text>{item}</Text>
              </li>
            );
          })}
        </ul>
      </div>
    );
  },
});
